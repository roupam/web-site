;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    SET_CURSOR
;;    DH=ROW, DL=COLUMN, BH=00 (PAGE NUMBER 0)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
SET_CURSOR    MACRO
        MOV    AH,02H
        MOV    BH,00H
        MOV    DH,#1
        MOV    DL,#2
        INT    10H
        #EM

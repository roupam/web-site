;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    MACRO LIBRARY FOR KEYBOARD FACILITIES
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
 
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    DISPLAY CHARACTER
;;    #1=CHARACTER CODE
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
DISPLAY    MACRO
        MOV    AH,02H
        MOV    DL,#1
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    PRINT CHAIN OF CHARACTERS
;;    #1=OFFSET OF MESSAGE
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
PRINT       MACRO
        MOV    DX,OFFSET #1
        MOV    AH,09H
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;      TO READ KEYB CHARACTER AND DISPLAY IT
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
READ        MACRO
        MOV    AH,01H
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    READ FROM FILE
;;      #1=NO OF BYTES TO READ
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
READ_F    MACRO
            MOV     CX,#1
            MOV     AH,3FH
            INT     21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    WRITE_TO_FILE
;;      #1=NO OF BYTES TO WRITE
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
WRITE_F    MACRO
        MOV    CX,#1
        MOV    AH,40H
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    OPEN_FILE
;;      #1=REQUEST CODE (00,01,02)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
OPEN_F    MACRO     
        MOV    AH,3DH
        MOV    AL,#1
            INT   21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    CLOSE_FILE
;;      
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
CLOSE_F    MACRO
            MOV     AH,3EH
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    CREATE_FILE
;;      NORMAL ATTRIBUTE        #1=00H
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
CREATE_F    MACRO
        MOV    AH,3CH
        MOV    CX,#1
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    MOVE FILE POINTER
;;      #1=00(START)01(CURRENT)02(END)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
FILE_PTR    MACRO
        MOV    AL,#1
        MOV    AH,42H
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;;;        #1=FILE_MASK
;;;;;;;;;;;;;;;;;;;;;;;;;;;
FIND_F      MACRO   #1
        MOV    AH,4EH
        MOV    CX,00H
        LEA    DX,#1
        INT    21H
        ENDM
 
FIND_NEXT    MACRO
        MOV    AH,4FH
        INT    21H
        ENDM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    UNLINK FILE
;;      #1=FILE ATTRIBUTES
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
UNLINK_F    MACRO
        MOV    AH,41H
            0MOV     CL,#1
        INT    21H
        #EM
 
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;    EXIT        #1=ERROR CODE
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 
EXIT        MACRO
        MOV    AH,4CH
        MOV    AL,#1
        INT    21H
        #EM
